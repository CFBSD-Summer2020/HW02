## `geom_smooth()` using formula 'y ~ x'

ggplot(iris, aes(Sepal.Length, Petal.Length, color = Species, shape = Species)) +
  geom_point() +
  geom_smooth(method = lm, se = FALSE, color = "black") +
  facet_wrap(vars(NewOrder), scales = "free_y")
